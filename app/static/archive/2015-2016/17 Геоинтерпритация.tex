\begin{task}
 Пусть $a,b,c,A,B,C>0$  и $a+A=b+B=c+C=k$. Докажите, что  $aB+bC+cA<k^2$.
\end{task}

\begin{task}
 Докажите, что если $a,b,c>0$, то $\sqrt{a^2-ab+b^2}+\sqrt{b^2-bc+c^2}\geqslant\sqrt{a^2+ac+c^2}$. Когда
достигается равенство?
\end{task}


\begin{task}
 Докажите, что \[\sqrt{a_1^2+b_1^2}+\dots+\sqrt{a_n^2+b_n^2}\geqslant \sqrt{(a_1+\dots+a_n)^2+(b_1+\dots+b_n)^2}.\]
Когда достигается равенство?
\end{task}

\begin{task}
 Пусть $|x_1|, |x_2|\leq 1$. Докажите неравенство \[\sqrt{1-x_1^2}+\sqrt{1-x_2^2}\leqslant 2\sqrt{1-\left(\frac{x_1+x_2}{2}\right)^2}.\]
Когда достигается равенство?
\end{task}

\begin{task}
\sub Докажите, что среди любых четырех чисел найдутся два числа $x, y$, такие что $0\leqslant \frac{x-y}{1+xy}\leqslant 1$.\\
\sub Докажите, что среди любых шести чисел найдутся два числа $x, y$, такие что $0\leqslant \frac{x-y}{1+xy}\leqslant \frac{1}{\sqrt{3}}$.
\end{task}


\begin{task}
 Если $0\leqslant x_1, x_2, x_3, x_4\leq 1$, то \[0\leqslant x_1(1-x_2)+x_2(1-x_3)+x_3(1-x_4)+x_4(1-x_1)\leqslant 2.\]
\end{task}


\begin{task}

Когда достигается равенство?
\end{task}
