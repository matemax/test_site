\begin{theorem}
Пусть $f$~--- непрерынвая на отрезке $[a,b]$  функция, такая, что $f(a)<0$ и $f(b)>0$. Докажите, что существует такая точка $x_0\in[a,b]$, что $f(x_0)=0$.
\end{theorem}

\begin{task}
Пусть $f$~--- непрерынвая на отрезке $[a,b]$. Докажите, что для любого $c\in[f(a),(b)]$ существует такая точка $x_0\in[a,b]$, что $f(x_0)=c$.
\end{task}

\begin{task}

\end{task}

\end{task}
