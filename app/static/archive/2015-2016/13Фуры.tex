По умолчанию мы находим функции $f:\RR\to\RR$.
Когда ты видишь ФУРу, то спроси себя:

\emph{Нет ли хорошей замены?}
\begin{task} $f(3x+5)=x^2+1$.
\end{task}
\begin{task} $f(x^2)=x+2$.
\end{task}
\emph{Не пишется ли <<обычная>> система уравнений?}

\begin{task} $x^2f(x)+f(1-x)=2x-x^4$.
\end{task}
\emph{Нельзя ли разделить переменные: $A(x)=B(y)$?}

\begin{task} $f(x)-f(y)=x^3-y^3$?
\end{task}
\emph{Нельзя ли отловить нарушение симметрии?}

\begin{task} $f(x+y)=x+f(y)$.
\end{task}
\emph{ ли посчитать $f(0), f(1), f(2), f(-1)$ и т.п и это использовать?}

\begin{task}$f(x)f(y)=f(xy)+x+y$.
\end{task}

\emph{ является ли функция инъективной?}

\begin{task} Докажите инъективность функции с условием $f(f(x)+f(y))=x+y$.
\end{task}
\begin{task} Найдите все инъективные функции со свойством $f(x+f(y))=f(f(x))+f(y)$.
\end{task}
\emph{Не является ли функция сюръективной?}

\begin{task} Докажите сюръективность функции с условием $f(f(x)+f(y))=x+y$.
\end{task}



\begin{task} $xf(y)+yf(x)=0$.
\end{task}
\begin{task} $f(\frac{x+1}{x})=1+\frac{1}{x}+\frac{1}{x^2}$ при $x\not=0$.
\end{task}
\begin{task} $f(1/x)+\frac{1}{x}f(-x)=x$ при $x\not=0$.
\end{task}
\begin{task}$f(f(x+y))=x+f(y)$.
\end{task}
\begin{task} $f(x-y)=f(x)+f(y)-2xy$.
\end{task}
\begin{task} $f(x-y)=f(x+y)f(y)$.
\end{task}
\begin{center}{\vspace{1pt}\large\bf Для решивших всё предыдущее}\end{center}\vspace{-10pt}

\begin{task} $f(x^2-y^2)=x^2-f(y^2)$. 
\end{task}
\begin{task} $f(x+y)-f(x-y)=4xy$. Указание: естественна замена.
\end{task}
\begin{task} $xf(y)+yf(x)=(x+y)f(x)f(y)$. Указание: не теряем случаи!
\end{task}
\begin{task} $f(x+y+f(y))=f(f(x))+2y$. Указание: сначала докажите инъективность.
\end{task}
