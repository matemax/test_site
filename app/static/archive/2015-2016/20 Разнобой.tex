\begin{task}
Пусть $a_n = \dfrac{1}{0!}+\dfrac{1}{1!}+\ldots+\dfrac{1}{n!}$.\\ \sub Докажите, что сущуествует предел $e=\lim\limits_{n\rightarrow\infty}a_n$.\\
\sub Докажите, что $e<3$.\\
\sub Пусть $b_n=\left(1+\dfrac{1}{n}\right)^n$. Докажите, что $b_n<a_n$\\
\sub Докажите, что существует предел $b =\lim\limits_{n\rightarrow\infty}b_n$.\\
\sub Докажите, что $b\leqslant e$.\\
\sub Докажите, что $b=e$.

\end{task}

\sub $\left(\dfrac{n}{4}\right)^n<n!<\left(\dfrac{n}{2}\right)^n$ при $n>1$;\\
\sub $\left(\dfrac{n}{e}\right)^n<n!<n\left(\dfrac{n}{e}\right)^n$ при $n>6$

\end{task}
