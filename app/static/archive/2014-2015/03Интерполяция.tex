\begin{task}
Существует ли многочлен $f(x)\in K[x]$, где $K$~--- поле, который в различных  точках $x_0,\ldots, x_n$ принимает данные значения $y_0=f(x_0),\ldots,y_n=f(x_n)$? Единственный ли этот многочлен?
\end{task}

\sub Докажите, что существует $n$ такое, что $n \mod_a 1$, $n \mod_b 0$.

\sub Докажите, что для любого $0\leqslant r_1 <a$ существует $n$ такое, что $n \mod_a r_1$, $n \mod_b 0$.

\sub Prove that for every  integer numbers $0\leqslant r_1 <a, 0\leqslant r_2 <b,$ there exist $n$ such that $n  \mod_a r_1$, $n  \mod_b r_2$.

\sub Докажите, что число $n$ из предыдущей задачи единственно по модулю $ab$.

\begin{task}[Китайская теорема об остатках] Пусть $m_1, m_2, \dots, m_k$ попарно взаимно простые числа, $m=m_1m_2\cdot \dots \cdot m_k$. Тогда для любых остатков $r_i$, $0 \leqslant r_i <m_i$ , существует единственный остаток $r$, $0\leqslant r <m$ такой, что $r \mod_{m_i} r_i$.
\end{task}

\begin{task}
Опишите все многочлены, которые в точках $x_0, x_1, \dots, x_n$ принимают значения $y_0, y_1, \dots, y_n$.
\end{task}

\begin{definition} Интерполяционный многочлен в виде
{\small \begin{align}\label{eq_Newton}
P(x)=a_0+a_1(x-x_0)+a_2(x-x_0)(x-x_1)+\ldots+a_n(x-x_0)(x-x_1)\cdot\ldots\cdot(x-x_{n-1})
\end{align}}
называется \emph{интерполяционным многочленом Ньютона}.
\end{definition}
\begin{task} Докажите, что для любых точек $x_0, x_1, \ldots, x_n$ и значений $y_0, y_1, \ldots, y_n$ существуют числа $a_i$ такие, что многочлен \eqref{eq_Newton} в точках $x_0, x_1, \dots, x_n$ принимает значения $y_0, y_1, \dots, y_n$.
\end{task}
