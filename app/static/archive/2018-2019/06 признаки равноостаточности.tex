\begin{task}

\begin{task}

\end{task}

\begin{task}

\begin{task}

\end{task}


\begin{task}

\begin{definition}
Знакопеременной суммой цифр числа $\overline{a_na_{n-1}\ldots a_2a_1}$ называется число $(-1)^{n-1}a_n+(-1)^{n-2}a_{n-1}+\ldots+a_3-a_2+a_1$.
\end{definition}
