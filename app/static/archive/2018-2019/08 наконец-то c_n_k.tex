\begin{task}
Найдите количество способов выбрать из $n$ различных шаров \sub 4, \sub $k$.
\end{task}

\begin{definition}
Количеством способов выбрать из $n$ различных предметов $k$ называется \emph{числом сочетаний из $n$ по $k$} и обозначается как $C_n^k$. \emph{Читается <<Цэ из эн по ка>>.}
\end{definition}
