\begin{definition}

\end{definition}


\begin{task}
Докажите свойства неравенств:\\
\sub если $a>b$ и $b>c$, то $a>c$;\\
\sub если $a+b>c$, то $a>c-b$;\\
\sub если $a>c$ и $b>d$, то $a+b>c+d$;\\
\sub если $a,b,c,d >0$, $a>c$ и $b>d$ , то $ab>cd$.
\end{task}
