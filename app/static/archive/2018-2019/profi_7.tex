\begin{task}
Докажите, что КСБ не более 100000.
\end{task}

\begin{definition} Обозначеним через $a_k$ -- количество трехзначных номеров с суммой цифр $k$,	 $b_k$ -- количество шестизначных номеров с суммой цифр $k$.
\end{definition}

\begin{task}
Докажите, что КСБ с суммой цифр $2k$ равно $a_k^2$.
\end{task}

\begin{task}
Найдите a) $a_4$; б) $a_9$.
\end{task}

\begin{task}
Докажите, что КСБ равно $a_0^2+a_1^2+\ldots+a_{27}^2$.
\end{task}

\begin{task}
Докажите, что $a_k= a_{27-k}$.
\end{task}

\begin{task}
Докажите, что КСБ равно $2(a_0^2+a_1^2+\ldots+a_{13}^2)$.
\end{task}

\begin{definition} Рассмотрим все тройки неотрицательных целых чисел, удовлетворяющих уравнению $x+y+z=k$. Назовем нарушением, если $x$, $y$ или $z$ больше 9. Назовем тройку хорошей, если в ней нет нарушений и плохой в противном случае. Аналогично определяются плохие и хорошие шестерки.
\end{definition}

\begin{task}
Найдите количество плохих троек при $k=10$ и $k=11$.
\end{task}

\begin{task}
Докажите, что при $10\leqslant k\leqslant 19$ количество плохих троек равно $3a_{k-10}$.
\end{task}

\begin{task}
Найдите все $a_k$ при $k=0,1,2,\ldots,12,13$ и вычислите КСБ.
\end{task}

\begin{task}

\end{task}

\begin{task}

\end{task}

\begin{task}
Докажите, что при $10\leqslant k$ количество плохих шестерок не более $C_{k-5}^5$.
\end{task}

\begin{task}
Докажите, что КСБ больше $C_{32}^5-6C_{22}^5$.
\end{task}

\begin{task}
Докажите, что при данном $k$ количество плохих шестерок с двумя нарушениями в данных местах равно $C_{k-15}^5$.
\end{task}

\begin{task}

\end{task}
