\begin{definition}
Идеалом в коммутативном кольце $K$, называется множество $I$, замкнутое относительно сложения и для любого $a\in K$ верно, что  $aI\subseteq I$, обозначается $I\triangleleft K$. 
\end{definition}


\begin{exercise}
\sub Привидите пример идеалов в кольце целых чисел. \sub Найдите все идеалы в поле.  \sub Привидите пример идеала в кольце $K[x]$.
\end{exercise}

\begin{definition}
Идеал  $I\triangleleft K$ называется порождённым, элементами $a_1,\ldots,a_n$, если $\forall a\in I$ существуют $b_1,\ldots,b_n\in K$, что $a=\sum b_ia_i$.
\end{definition}


\begin{definition}
Идеал $I\triangleleft K$ называется главным, если существет $k\in I$ такой, что $\forall i\in I$ верно, что $i~\vdots~k$. Обозначается $I=(k)$.
\end{definition}

\begin{exercise}
Привидите пример неглавного идеала в $K[x,y]$.
\end{exercise}

\begin{task}
Докажите, что все идеалы в кольце целых чисел и кольце многочленов от одной переменной над полем главные.
\end{task}

\begin{definition}
Радикалом идеала $I\triangleleft K$ называется множество элементов $J\subseteq K$, что $\forall j \in J~\exists m\in \mathbb{N}$, что $j^m\in I$, обозначается $J=\sqrt{I}$. 
\end{definition}

\begin{exercise}
Найдите радика идеала $I=((x-1)^2(x-2)^3)\triangleleft \mathbb{R}[x]$.
\end{exercise}

\begin{task}
Докажите, что радикал идеала тоже идеал.
\end{task}

\begin{definition}
Пусть $I\triangleleft K$, тогда два элемента $a,b\in K$ называются сравнимыми по модулю идеала, если $a-b \in I$. Обозначается $a\mod_{I} b$.
\end{definition}


\begin{definition}
Многообразием идеала $I\triangleleft K[x_1,x_2,\ldots x_n]$, где $K$ поле, назвается множество точек $V \subseteq K^n$ таких, что $\forall x_0\in V$ и $\forall f\in I$ верно, что $f(x_0)=0$, обозначается $V(I)$.

Пусть $V'\subseteq K^n$, тогда идеалом, порождённым $V'$ называется множество всех  многочленов в $f\in K[x_1,x_2,\ldots x_n]$ таких, что $\forall x_0\in V'$ верно, что $f(x_0)=0$, обозначается $I(V')$
\end{definition}

\begin{theorem}{\bf теорема Гильберта о нулях.}
Для любого алгебраически замкнутого поля $K$ и идеала $I\triangleleft K[x_1,x_2,\ldots x_n]$ верно, что $\sqrt{I}=I(V(I))$.
\end{theorem}
\begin{theorem}{\bf теорема Гильберта о нулях, слабая версия.}
Для любого алгебраически замкнутого поля $K$ и идеала $I\triangleleft K[x_1,x_2,\ldots x_n]$ верно, что $I=K\Leftrightarrow V(I)=\emptyset$.
\end{theorem}
